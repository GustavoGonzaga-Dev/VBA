Sub JAPA()

    Dim Num As Integer
    Dim Aux As Integer
    Dim Rev As Integer
    
    Num = InputBox(Prompt:="Insira um numero natural: ")
    
    If (Num > 1 And Num < 10000) Then
    
    Aux = Num
    Rev = 0
    
    If (Aux <> 0) Then
        
        Do
        Rev = ((Rev * 10) + (Aux Mod 10))
        Aux = Aux / 10
        
        
        Loop While Rev < Num
            If (Rev = Num) Then
                MsgBox Prompt:="" & Rev & " é um palindromo"
            Else
                MsgBox Prompt:="" & Rev & " não é um palindromo"
            End If
         
    End If
    
    Else
     MsgBox Prompt:="Numero invalido"
    End If
    
End Sub
