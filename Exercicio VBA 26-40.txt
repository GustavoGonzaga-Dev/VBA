Sub ex27()

    Dim NVoltas As Integer
    Dim Tempo As Integer
    Dim Metros As Integer
    Dim Vm As Double
    Dim Res As Double
    
    NVoltas = InputBox(Prompt:="Insira o numero de voltas: ")
    Tempo = InputBox(Prompt:="Insira o tempo do percurso em minutos: ")
    Metros = InputBox(Prompt:="Insira o tamanho do percurso em metros: ")
    
    Res = Tempo / NVoltas
    Vm = Metros / Res
    
    MsgBox Prompt:="A velocidade media por volta foi de: " & Res & "Km/h"

Sub ex29()

    Dim ValorInv As Double
    Dim Valido As Integer
    Dim Des As Double
    
    
    Valido = InputBox(Prompt:="Insira:" & vbCrLf & "1- Conta poupança" & vbCrLf & "2- Renda fixa")
    
    Select Case Valido
        Case 1
            ValorInv = InputBox(Prompt:="Insira o valor do investimento: R$")
            Des = ((ValorInv * 3) / 100)
            ValorInv = ValorInv + Des
            MsgBox Prompt:="Seu saldo após 30 dias é de: " & ValorInv
        Case 2
            ValorInv = InputBox(Prompt:="Insira o valor do investimento: R$")
            Des = ((ValorInv * 5) / 100)
            ValorInv = ValorInv + Des
            MsgBox Prompt:="Seu saldo após 30 dias é de: R$" & ValorInv
        Case Else
            MsgBox Prompt:="Valor inserido não permitido."
    End Select
    
End Sub
